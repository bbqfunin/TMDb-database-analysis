def movietype(df):
    mtype = []
    for movielist in df['genres']:
        mtype = mtype + movielist
    return mtype, list(set(mtype)) # dfdf

def tcounts(list1,list2): #list1list2list1
    typecounts = {}
    typecounts = typecounts.fromkeys(list2,0)
    for i in list1:
        typecounts[i] += 1
    return typecounts

keylist = set(list(h_typecounts.keys()) + list(m_typecounts.keys()) + list(l_typecounts.keys())) # 
def add0(df): #keylist0
    for i in keylist:
        if i not in df.keys():
            df[i] = 0
    return df

def sortbykey(df): #
    sortdict = {}
    for k in sorted(df.keys()):
        sortdict[k] = df[k]
    return sortdict

h_typecounts = sortbykey(add0(h_typecounts))
m_typecounts = sortbykey(add0(m_typecounts))
l_typecounts = sortbykey(add0(l_typecounts))
